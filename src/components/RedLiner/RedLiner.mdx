import { Playground, Props } from 'docz';
import RedLiner from './RedLiner';
import { buttonStyle, inputStyle } from '../../utils/exampleStyles.ts'

# RedLiner

The RedLiner component can be used to show style attritbutes on any display component.

## Props

<Props of={RedLiner} />

## Usage

### Default Options

The default options.

<Playground>
    <RedLiner>
      <button style={buttonStyle}>
        Measure Me
      </button>
    </RedLiner>
</Playground>

### InfoBox Options

Specify which attributes to show in the InfoBox.

<Playground>
  <RedLiner infoOpts={['borderRadius', 'color']}>
    <button style={buttonStyle}>Measure Me</button>
  </RedLiner>
</Playground>

### Show on Hover

Use the prop `showOnHover` to hide RedLiner until the child component is hovered.
<Playground>
  <RedLiner showOnHover>
    <button style={buttonStyle}>helllllo</button>
  </RedLiner>
</Playground>

### Change color

The `color` prop allows specifying the line color.

<Playground>
  <RedLiner color="blue">
    <input style={inputStyle}></input>
  </RedLiner>
</Playground>
