import { Playground, Props } from 'docz';
import RedLiner from './RedLiner';
import { buttonStyle, inputStyle } from '../../utils/exampleStyles.ts'

# RedLiner

Used to show style attritbutes on any display component.

## Props

<Props of={RedLiner} />

## Usage

### Default Options

The default options.

<Playground>
    <RedLiner>
      <button style={buttonStyle}>
        Measure Me
      </button>
    </RedLiner>
</Playground>

### Config Prop

Customize color of lines, sections to display, and properties to show in the InfoBox.

<Playground>
  <RedLiner config={{ color: 'blue', infoOpts: ['borderRadius', 'color'] }}>
    <button style={buttonStyle}>Measure Me</button>
  </RedLiner>
</Playground>

<br />

Remove the InfoBox component with `displayOpts`.

<Playground>
  <RedLiner config={{ displayOpts: ['height', 'width'] }}>
    <input style={inputStyle}></input>
  </RedLiner>
</Playground>


### Show on Hover

Use the prop `showOnHover` to hide RedLiner until the child component is hovered.
<Playground>
  <RedLiner showOnHover>
    <button style={buttonStyle}>helllllo</button>
  </RedLiner>
</Playground>

### Maximum Customization

<Playground>
  <RedLiner config={{ color: 'purple', displayOpts: ['height', 'info'], infoOpts: ['borderRadius', 'color'] }} showOnHover>
    <button style={buttonStyle}>Measure Me</button>
  </RedLiner>
</Playground>
